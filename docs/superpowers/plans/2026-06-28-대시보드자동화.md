> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 세션 현황판에 "새 세션 시작" 폼을 추가하고, 입력값(상태+재료)으로부터 Claude가 재료·질문지·대본·AI티제거를 자동 생성하며, 최종 결과물을 현황판에서 실시간으로 확인 가능하게 한다.

**Architecture:** 
- 프론트엔드: 현황판.html에 모달 폼 추가 (상태 드롭다운 + 재료 텍스트)

- 자동화: Claude MCP 호출로 ②~⑤ 단계 자동 생성
- 현황판: 폼 제출 후 30초 내 재생성 (automations/session-dashboard.mjs 재실행)
- 검수: 승인 버튼으로 사람이 최종 확인 후 다음 단계 진행

**Tech Stack:**
- Frontend: HTML5 + vanilla JS (현황판.html 수정)
- Backend: Node.js + Express (하루끝-키트/사이트_레벨2/api/)
- Database: Supabase PostgreSQL
- AI: Claude MCP (Anthropic SDK)
- Automation: Node.mjs 스크립트 (automations/session-generator.mjs)

```sql
-- 대기 중인 세션들을 관리하는 큐
CREATE TABLE sessions_queue (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  status TEXT NOT NULL DEFAULT 'created', -- created, generating, generated, approved, archived
  state_name TEXT NOT NULL, -- '텅빈저녁', '멍한주말오후' 등
  materials_text TEXT NOT NULL, -- 사용자가 입력한 재료(자막·기사·메모)
  created_at TIMESTAMP DEFAULT NOW(),
  generation_started_at TIMESTAMP,
  generation_completed_at TIMESTAMP,
  approved_at TIMESTAMP,
  approver_notes TEXT,
  storage_path TEXT -- outputs 폴더 내 저장 경로
);

-- 각 단계별 생성된 스크립트들 (②~⑤)
CREATE TABLE session_scripts (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  session_id UUID NOT NULL REFERENCES sessions_queue(id) ON DELETE CASCADE,
  stage INT NOT NULL, -- 2(재료), 3(질문지), 4(대본), 5(AI티제거)
  stage_name TEXT NOT NULL,
  content TEXT NOT NULL,
  version INT DEFAULT 1,
  status TEXT DEFAULT 'pending', -- pending, approved, rejected
  created_at TIMESTAMP DEFAULT NOW(),
  updated_at TIMESTAMP DEFAULT NOW()
);

-- RLS 활성화
ALTER TABLE sessions_queue ENABLE ROW LEVEL SECURITY;
ALTER TABLE session_scripts ENABLE ROW LEVEL SECURITY;

-- 익명 사용자도 읽을 수 있도록
CREATE POLICY "sessions_queue_select" ON sessions_queue FOR SELECT USING (TRUE);
CREATE POLICY "session_scripts_select" ON session_scripts FOR SELECT USING (TRUE);
```

- [ ] **Step 3: API 엔드포인트 구현**

```javascript
import { createClient } from '@supabase/supabase-js';

const supabase = createClient(
  process.env.SUPABASE_URL,
  process.env.SUPABASE_ANON_KEY
);

export default async function handler(req, res) {
  // CORS + OPTIONS 처리
  if (req.method === 'OPTIONS') {
    return res.status(200).end();
  }

  if (req.method !== 'POST') {
    return res.status(405).json({ error: 'Method not allowed' });
  }

  try {
    const { stateName, materialsText } = req.body;

    // 입력 검증
    if (!stateName || !materialsText) {
      return res.status(400).json({
        error: 'stateName과 materialsText는 필수입니다.',
      });
    }

    if (materialsText.trim().length < 10) {
      return res.status(400).json({
        error: '재료는 최소 10자 이상이어야 합니다.',
      });
    }

    // Supabase 삽입
    const { data, error } = await supabase
      .from('sessions_queue')
      .insert([
        {
          state_name: stateName,
          materials_text: materialsText.trim(),
          status: 'created',
        },
      ])
      .select()
      .single();

    if (error) {
      console.error('Supabase insert error:', error);
      return res.status(500).json({ error: 'Database error' });
    }

    return res.status(201).json({
      success: true,
      sessionId: data.id,
      message: '세션이 생성되었습니다. 자동 생성 중...',
    });
  } catch (err) {
    console.error('Unhandled error:', err);
    return res.status(500).json({ error: 'Internal server error' });
  }
}
```

```javascript
const OUTPUT_BASE = join(process.cwd(), '../../outputs');

export default async function handler(req, res) {
  if (req.method !== 'POST') {
    return res.status(405).json({ error: 'Method not allowed' });
  }

  try {
    const { sessionId } = req.body;

    if (!sessionId) {
      return res.status(400).json({ error: 'sessionId is required' });
    }

    // Supabase에서 세션 정보 조회
    const { data: session, error: selectError } = await supabase
      .from('sessions_queue')
      .select('id, state_name, materials_text')
      .eq('id', sessionId)
      .single();

    if (selectError || !session) {
      return res.status(404).json({ error: 'Session not found' });
    }

    // 상태 업데이트: generating
    await supabase
      .from('sessions_queue')
      .update({ status: 'generating', generation_started_at: new Date() })
      .eq('id', sessionId);

    const results = {};

    try {
      // ② 재료 추출
      console.log(`[${sessionId}] Generating materials...`);
      const materials = await generateMaterials(
        session.materials_text,
        session.state_name
      );
      results.materials = materials;

      await supabase.from('session_scripts').insert([{
        session_id: sessionId,
        stage: 2,
        stage_name: '재료 모으기',
        content: materials,
        status: 'pending',
      }]);

      // ③ 질문지 생성
      console.log(`[${sessionId}] Generating questions...`);
      const questions = await generateQuestions(materials, session.state_name);
      results.questions = questions;

      await supabase.from('session_scripts').insert([{
        session_id: sessionId,
        stage: 3,
        stage_name: '질문지·과제',
        content: questions,
        status: 'pending',
      }]);

      // ④ 대본 작성
      console.log(`[${sessionId}] Generating script...`);
      const script = await generateScript(session.state_name, questions);
      results.script = script;

      await supabase.from('session_scripts').insert([{
        session_id: sessionId,
        stage: 4,
        stage_name: '대본 초안',
        content: script,
        status: 'pending',
      }]);

      // ⑤ AI티 제거
      console.log(`[${sessionId}] Removing AI tone...`);
      const polished = await removeAITone(script);
      results.polished = polished;

      await supabase.from('session_scripts').insert([{
        session_id: sessionId,
        stage: 5,
        stage_name: 'AI티 빼기',
        content: polished,
        status: 'pending',
      }]);

      // 파일 저장
      const sessionDir = join(OUTPUT_BASE, sessionId);
      mkdirSync(sessionDir, { recursive: true });

      writeFileSync(join(sessionDir, '01_재료.md'), materials);
      writeFileSync(join(sessionDir, '02_질문지.md'), questions);
      writeFileSync(join(sessionDir, '03_대본_원본.md'), script);
      writeFileSync(join(sessionDir, '04_대본_윤문.md'), polished);

      // 상태 업데이트: generated
      await supabase
        .from('sessions_queue')
        .update({
          status: 'generated',
          generation_completed_at: new Date(),
          storage_path: sessionDir,
        })
        .eq('id', sessionId);

      return res.status(200).json({
        success: true,
        sessionId,
        stage: 5,
        message: '자동 생성 완료. 검수 대기 중...',
      });
    } catch (genError) {
      console.error(`[${sessionId}] Generation failed:`, genError);
      await supabase
        .from('sessions_queue')
        .update({ status: 'error' })
        .eq('id', sessionId);
      throw genError;
    }
  } catch (err) {
    console.error('Unhandled error:', err);
    return res.status(500).json({ error: 'Internal server error' });
  }
}
```

```javascript
  try {
    // Step 1: 세션 생성 API 호출
    const createRes = await fetch('/api/sessions/create', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ stateName, materialsText }),
    });

    if (!createRes.ok) {
      const error = await createRes.json();
      throw new Error(error.error || 'Failed to create session');
    }

    const { sessionId } = await createRes.json();

    statusDiv.textContent = '자동 생성 중... (2~3분 소요)';

    // Step 2: 자동 생성 API 호출 (백그라운드에서)

          headers: { 'Content-Type': 'application/json' },

        });
```

```javascript
async function fetchSessionsData() {
  try {
    const { data: sessions, error } = await supabase
      .from('sessions_queue')
      .select(`
        id,
        state_name,
        status,
        created_at,
        generation_started_at,
        generation_completed_at,
        approved_at,
        session_scripts(stage, stage_name, status, content)
      `)
      .order('created_at', { ascending: false })
      .limit(50);

    if (error) throw error;

    return sessions.map(s => ({
      id: s.id,
      stateName: s.state_name,
      status: s.status,
      createdAt: new Date(s.created_at),
      scripts: s.session_scripts || [],
      approvedAt: s.approved_at,
    }));
  } catch (err) {
    console.error('Failed to fetch sessions:', err);
    return [];
  }
}

function calculateProgress(scripts) {
  // 완성된 단계 수 / 7 (전체 단계)
  const completed = scripts.filter(s => s.status === 'approved').length;
  return Math.ceil((completed / 7) * 100);
}
```

```javascript
const supabase = createClient(
  process.env.SUPABASE_URL,
  process.env.SUPABASE_ANON_KEY
);

export default async function handler(req, res) {
  const { id } = req.query;

  if (!id) {
    return res.status(400).json({ error: 'Session ID required' });
  }

  try {
    const { data: session, error: sessionError } = await supabase
      .from('sessions_queue')
      .select('*')
      .eq('id', id)
      .single();

    if (sessionError || !session) {
      return res.status(404).json({ error: 'Session not found' });
    }

    const { data: scripts, error: scriptError } = await supabase
      .from('session_scripts')
      .select('*')
      .eq('session_id', id)
      .order('stage', { ascending: true });

    if (scriptError) throw scriptError;

    return res.status(200).json({
      session,
      scripts: scripts || [],
    });
  } catch (err) {
    console.error('Error:', err);
    return res.status(500).json({ error: 'Internal server error' });
  }
}
```

```javascript
const supabase = createClient(
  process.env.SUPABASE_URL,
  process.env.SUPABASE_ANON_KEY
);

export default async function handler(req, res) {
  if (req.method !== 'POST') {
    return res.status(405).json({ error: 'Method not allowed' });
  }

  const { id } = req.query;

  try {
    const { error } = await supabase
      .from('sessions_queue')
      .update({
        status: 'approved',
        approved_at: new Date(),
      })
      .eq('id', id);

    if (error) throw error;

    // 모든 scripts를 'approved'로 갱신
    await supabase
      .from('session_scripts')
      .update({ status: 'approved' })
      .eq('session_id', id);

    return res.status(200).json({
      success: true,
      message: '승인 완료. 녹음 단계로 진행합니다.',
    });
  } catch (err) {
    console.error('Error:', err);
    return res.status(500).json({ error: 'Internal server error' });
  }
}
```

```javascript
import { createClient } from '@supabase/supabase-js';

const BASE_URL = process.env.VERCEL_URL || 'http://localhost:3000';

  process.env.SUPABASE_URL,

);

describe('Dashboard Automation Integration', () => {
  let sessionId;

  it('전체 파이프라인 실행: 입력 → 생성 → 검수 → 승인', async () => {
    // 1. 세션 생성
    const createRes = await fetch(`${BASE_URL}/api/sessions/create`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },

        stateName: '통합테스트',

      }),
    });

    expect(createRes.status).toBe(201);
    const { sessionId: id } = await createRes.json();
    sessionId = id;

    // 2. 자동 생성 실행 (잠시 대기 후)
    await new Promise(r => setTimeout(r, 2000));

    const generateRes = await fetch(`${BASE_URL}/api/sessions/generate`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ sessionId }),
    });

    expect(generateRes.status).toBe(200);

    // 3. 세션 상태 확인
    const { data: session } = await supabase
      .from('sessions_queue')
      .select('status')
      .eq('id', sessionId)
      .single();

    expect(session.status).toBe('generated');

    // 4. 승인 처리
    const approveRes = await fetch(
      `${BASE_URL}/api/sessions/${sessionId}/approve`,
      { method: 'POST' }
    );

    expect(approveRes.status).toBe(200);

    // 5. 최종 상태 확인
    const { data: finalSession } = await supabase
      .from('sessions_queue')
      .select('status')
      .eq('id', sessionId)
      .single();

    expect(finalSession.status).toBe('approved');
  });

  afterAll(async () => {
    // 테스트 데이터 정리

      await supabase
        .from('sessions_queue')
        .delete()
        .eq('id', sessionId);
    }
  });
```
